function highAndLow(numbers){
         var arr = numbers.split(" ");
         var min = parseInt(arr[0]);
         var max = parseInt(arr[0]);
         for(var i = 1; i < arr.length; i++){
             var tmp = parseInt(arr[i]);
             if(tmp < min) min = tmp;
             if(tmp > max) max = tmp;
         }
         return max + " " + min;
    }
